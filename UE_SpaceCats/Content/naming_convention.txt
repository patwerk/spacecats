naming_convention_for_art_assets:

materials
	m_assetn_x_y
	(Material_AssetName_VariantExtension)
	(Example: m_plantpot_a_01)
	
meshes
	actors (skeletal meshes)
		cat_base
		cat_mods
			a_assetn_x_y
			(Actor_AssetName_VariantExtension)
			(Example: a_cat_a_base)
			[Reading backwards as: Base for Cat "A", who's an Actor]
			(Example: a_cat_a_tail_01)
			[Reading backwards as: First tail for Cat "A", who's an Actor]
			
	static (static meshes)
		furniture
		ingredients
		planet
		plants
			sm_assetn_x_y
			(StaticMesh_AssetName_VariantExtension)
			(Example: sm_cupboard_a)
			(Example: sm_floor)
		
textures
		t_assetn_x_y
		(Texture_AssetName_VariantExtension)
		(Example: t_cupboard_a_01)
		[Reading backwards as: First Texture Variation 01 of Cupboard "A" Texture]
		(Example: t_cat_a_base_01_n)
		[Reading backwards as: NormalMap Texture Variation 01 of Base for Cat "A"]
